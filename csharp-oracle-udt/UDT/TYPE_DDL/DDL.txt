CREATE OR REPLACE TYPE TEST.Barcode AS OBJECT
(
  Data VARCHAR2(20),
  Type VARCHAR2(20),
  Length NUMBER
)
/

CREATE OR REPLACE TYPE TEST.BARCODEARRAY
AS VARRAY (1000) OF BARCODE
/


CREATE OR REPLACE TYPE TEST.BCRDATA AS OBJECT
(
  Id VARCHAR2(20),
  Count NUMBER,
  Height NUMBER,
  Barcodes TEST.BARCODEARRAY
)
/
